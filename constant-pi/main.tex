    \frac {c}{C}
        &= \lim_{\alpha \to \infty} \frac{
            \frac{360}{\alpha} k
        }{
            \frac{360}{\alpha} K
        } &&\\\nonumber
        &= \frac{k}{K} &&\\\nonumber
        &= \frac{r}{R} &&\\
    \Rightarrow \frac{c}{r} &= \frac{C}{R} = const &&\\
    \Rightarrow \pi &= \frac{c}{2r} = const
\end{flalign}
