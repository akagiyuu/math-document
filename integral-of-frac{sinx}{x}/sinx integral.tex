\begin{flalign}
\int \sin x dx
&= \int \Im\left(e^{ix}\right)dx\\\nonumber
&= \Im \left(\int e^{ix} dx\right)\\\nonumber
&= \Im\left(\frac{e^{ix}}{i}\right)\\\nonumber
&= \Im\left(-i\cos x + sinx\right)\\\nonumber
&= -\cos x
\end{flalign}
